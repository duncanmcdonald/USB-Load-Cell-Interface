516 inline void flip_sample_packet(struct sample_packet *cur, U8 cur_index)
517  {
518  U8 sreg;
519  U32 curtimer_value;
520 
521  // Grab the next packet number
522  last_packet_number += 1;
523 
524  // Disable ints
525  sreg = SREG;
526  cli();
527 
528  // Remember the tick count (with ints disabled)
529  curtimer_value = timer_value;
530 
531  if (lock_sample_packet_index != 0xff)
532  {
533  // There is a read-lock indicating overflow. Must discard this packet
534  cur -> sample_count = 0; //equivalent to sample_packet.sample_count; "struct name"."column of struct"
535 
536  // Enable ints and return - next sample will be written to this buffer
537  SREG = sreg;
538 
539  debug_write_string("PKOF!\n");
540  return;
541  } else
542  {
543  // There is not a read-lock present. While ints disabled, clear the
544  // can_read_sample_packet value so that a reader will not pick this
545  // up while we are finishing it
546  can_read_sample_packet_index = 0xff;
547 
548  // Flip to the next buffer for next sample
549  if (cur_index == 0) write_sample_packet_index=1;
550  else write_sample_packet_index = 0;
551 
552  /* Reset so that we start over on the next go round. sample_packets is an array of two structs sample_packet.
553  Only two elements in the array so write_sample_packet_index can only be 0 or 1*/
554  sample_packets[write_sample_packet_index].sample_count = 0;
555 
556  // Enable ints
557  SREG = sreg;
558 
559  // Finish this packet
560  cur -> packet_deltaticks = curtimer_value - last_packetticks; //LHS equivalent to sample_packet.packet_deltaticks
561  last_packetticks = curtimer_value;
562  cur -> packet_number = last_packet_number; //LHS equivalent to sample_packet.packet_number
563  cur -> control = 0xff; //LHS equivalent to sample_packet.control; this could be used for the Record ID
564 
565  // Mark the packet as readable
566  can_read_sample_packet_index = cur_index;
567  }
568  }
