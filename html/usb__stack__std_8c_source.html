360 void usb_stack_std_get_descriptor(U8 requestType)
361  {
362  data_buffer_t buffer;
363 
364  U16 length;
365  U16 offset;
366  U8 descriptortype;
367  U8 stringtype;
368  U8 dummy;
369  bit zlp;
370  U8 cur_cnt;
371 
372  stringtype = Usb_read_byte(); //Field = wValue lower byte; descriptor index
373  descriptortype = Usb_read_byte(); //Field = wValue upper byte; descriptor type
374 
375  usbstack_debug_write_string("SGD ");
376  usbstack_debug_write_hex(stringtype);
377  usbstack_debug_write_hex(descriptortype);
378 
379  buffer.type = 0;
380  buffer.length = 0;
381  usb_stack_user_populate_descriptor(buffer, requestType, descriptortype, stringtype);
382 
383  // No descriptor found?
384  if (buffer.type == 0)
385  {
386  Usb_enable_stall_handshake();
387  Usb_ack_receive_setup();
388  return;
389  }
390 
391  dummy = Usb_read_byte(); // wIndex field not needed
392  dummy = Usb_read_byte();
393 
394  LSB(length) = Usb_read_byte();
395  MSB(length) = Usb_read_byte();
396  Usb_ack_receive_setup(); // ack setup stage
397 
398  if (length > buffer.length)
399  {
400  if ((buffer.length % EP_CONTROL_LENGTH) == 0) zlp = TRUE; // Can send the whole thing
401  else zlp = FALSE;
402  } else
403  {
404  buffer.length = length; // Send only requested amount
405  zlp = TRUE;
406  }
407 
408  offset = 0;
409  while (buffer.length != 0 && !Is_usb_receive_out())
410  {
411  while (!Is_usb_read_control_enabled());
412 
413  cur_cnt = 0;
414  while (buffer.length != 0)
415  {
416  if (cur_cnt == EP_CONTROL_LENGTH) break; // New packet
417  cur_cnt += 1;
418 
419  Usb_write_byte(data_buffer_read_byte(buffer, offset));
420  offset += 1;
421  buffer.length -= 1;
422  }
423 
424  Usb_send_control_in(); // ack data stage; Clear Transmitter (IN) Ready Interrupt Flag
425  }
426 
427  if (Is_usb_receive_out())
428  {
429  Usb_ack_receive_out();
430  return;
431  } // Abort from host
432  if (zlp == TRUE)
433  {
434  while (!Is_usb_read_control_enabled());
435  Usb_send_control_in(); // Clear Transmitter (IN) Ready Interrupt Flag
436  }
437 
438  while (!Is_usb_receive_out()); // wait to receive OUT data
439  Usb_ack_receive_out(); // ack status transaction
440  }
