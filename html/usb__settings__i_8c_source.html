158 static inline void user_populate_descriptor(data_buffer_t* data_buffer, U8 requesttype, U8 descriptortype, U8 stringtype) {
159  data_buffer->type=DBTYPE_LPM;
160 
161  if (requesttype==REQUEST_DEVICE_STATUS) {
162  // Handle device level descriptor requests
163  switch (descriptortype) {
164  case DEVICE_DESCRIPTOR:
165  data_buffer->length=sizeof(usb_dev_desc);
166  data_buffer->pointer.lpm=(PGM_P)&usb_dev_desc;
167  return;
168  case CONFIGURATION_DESCRIPTOR:
169  data_buffer->length=sizeof(usb_conf_desc);
170  data_buffer->pointer.lpm=(PGM_P)&usb_conf_desc;
171  return;
172  }
173 
174  // Other descriptor?
175  if (descriptortype==STRING_DESCRIPTOR) {
176  switch (stringtype) {
177  case LANG_ID:
178  data_buffer->length=sizeof(usb_user_language_id);
179  data_buffer->pointer.lpm=(PGM_P)&usb_user_language_id;
180  return;
181  case MAN_INDEX:
182  data_buffer->length=sizeof(usb_user_manufacturer_string_descriptor);
183  data_buffer->pointer.lpm=(PGM_P)&usb_user_manufacturer_string_descriptor;
184  return;
185  case PROD_INDEX:
186  data_buffer->length=sizeof(usb_user_product_string_descriptor);
187  data_buffer->pointer.lpm=(PGM_P)&usb_user_product_string_descriptor;
188  return;
189  case SN_INDEX:
190  data_buffer->length=sizeof(usb_user_serial_number);
191  data_buffer->pointer.lpm=(PGM_P)&usb_user_serial_number;
192  return;
193  }
194  }
195  } else if (requesttype==REQUEST_INTERFACE_STATUS) {
196  // Handle requests for interface descriptors
197  switch (descriptortype) {
198  case REPORT:
199  data_buffer->length = sizeof(usb_hid_report_descriptor);
200  data_buffer->pointer.lpm=(PGM_P)&(usb_hid_report_descriptor[0]);
201  return;
202 
203  case HID:
204  data_buffer->length = sizeof(usb_conf_desc.hid);
205  data_buffer->pointer.lpm = (PGM_P)&(usb_conf_desc.hid.bLength);
206  return;
207  }
208  }
209 
210  data_buffer->type=0;
211 }
